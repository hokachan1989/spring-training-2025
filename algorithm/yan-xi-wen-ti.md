# --- 2人対戦じゃんけんゲーム (解答例) ---

print("じゃんけんゲーム！")
print("1: グー")
print("2: チョキ")
print("3: パー")

player1_choice = int(p1_choice_str)

# 2. プレイヤー2の手を入力
p2_choice_str = input("プレイヤー2さん、手を選んでください (1-3): ")
player2_choice = int(p2_choice_str)

# プレイヤーの手を表示
# （入力された数字に対応する手を表示するとより親切）
p1_hand = ""
if player1_choice == 1:
    p1_hand = "グー"
elif player1_choice == 2:
    p1_hand = "チョキ"
elif player1_choice == 3:
    p1_hand = "パー"
else:
    p1_hand = "無効な手" # 入力チェックがない場合の考慮

p2_hand = ""
if player2_choice == 1:
    p2_hand = "グー"
elif player2_choice == 2:
    p2_hand = "チョキ"
elif player2_choice == 3:
    p2_hand = "パー"
else:
    p2_hand = "無効な手" # 入力チェックがない場合の考慮

print(f"プレイヤー1: {p1_hand}({player1_choice}), プレイヤー2: {p2_hand}({player2_choice})")


# 3. 勝敗判定 & 4. 結果表示
# (入力が1,2,3以外の場合のエラー処理は省略)

    print("結果: あいこです！")
elif (player1_choice == 1 and player2_choice == 2) or \

     (player1_choice == 3 and player2_choice == 1):

    print("結果: プレイヤー2の勝ち！")
